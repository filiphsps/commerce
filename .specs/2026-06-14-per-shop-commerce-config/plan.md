# Per-shop commerce config Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make catalog page size, the geo-redirect dismissal TTL, and per-currency free-shipping messaging thresholds per-shop configurable via the existing `commerce` settings group.

**Architecture:** The shop shape is dual-source-of-truth: a CMS descriptor (`content-shapes.ts`, drives editor + storefront types via `pnpm cms:gen`) plus three hand-written mirrors (`ShopBase`, `shopCommerceValidator`, `COMMERCE_DEFAULTS`). Each new `commerce.*` field is added to all four and gated by `cms:gen:check`. Storefront reads `shop.commerce?.x ?? fallback` so unset shops are byte-identical to today.

**Tech Stack:** TypeScript, Next.js 16 (Server Components), Convex (`convex/values` validators), Biome, Vitest, gql.tada. Build packages before lint/typecheck/test: `pnpm build:packages`.

```ts
/**
 * Clamps a requested catalog page size into Shopify's valid `first` range.
 *
 * @param value - Requested page size; may be a shop-config override or a per-surface fallback.
 * @param fallback - Page size used when `value` is not a finite number; defaults to `35`.
 * @returns An integer in `[1, 250]` (Shopify's max `first`).
 */
export const clampPageSize = (value: number, fallback = 35): number => {
    const n = Number.isFinite(value) ? value : fallback;
    return Math.min(250, Math.max(1, Math.floor(n)));
};
```

```ts
export type FreeShippingThreshold = { currencyCode: string; amount: number };

export type FreeShippingState =
    | { state: 'none' }
    | { state: 'progress'; threshold: number; remaining: number }
    | { state: 'unlocked'; threshold: number; remaining: 0 };

/**
 * Resolves the free-shipping messaging state for a cart against the shop's per-currency thresholds.
 * Display-only: Shopify still computes real shipping at checkout.
 *
 * @param input.thresholds - Per-currency free-shipping thresholds from `shop.commerce`.
 * @param input.currencyCode - The cart's presentment currency.
 * @param input.subtotal - The cart subtotal as a number in that currency.
 * @returns `none` when no threshold applies or the cart is empty; otherwise `progress` (with the
 *   remaining amount) or `unlocked`.
 */
export const resolveFreeShipping = ({
    thresholds,
    currencyCode,
    subtotal,
}: {
    thresholds: FreeShippingThreshold[] | undefined;
    currencyCode: string;
    subtotal: number;
}): FreeShippingState => {
    if (subtotal <= 0) {
        return { state: 'none' };
    }

    const match = thresholds?.find((t) => t.currencyCode === currencyCode);
    if (!match || !(match.amount > 0)) {
        return { state: 'none' };
    }

    if (subtotal >= match.amount) {
        return { state: 'unlocked', threshold: match.amount, remaining: 0 };
    }

    return { state: 'progress', threshold: match.amount, remaining: match.amount - subtotal };
};
```

- [ ] **Step 3: Implement `FreeShippingProgress`**

Create `apps/storefront/src/components/cart/free-shipping-progress.tsx`:

```tsx
'use client';

import { Price } from '@/components/products/price';
import { type CurrencyCode, getTranslations, type LocaleDictionary } from '@/utils/locale';
import { cn } from '@/utils/tailwind';
import type { FreeShippingState } from './free-shipping';

export type FreeShippingProgressProps = {
    state: FreeShippingState;
    currencyCode: CurrencyCode;
    i18n: LocaleDictionary;
};

/**
 * Free-shipping merchandising line for the cart summary: a progress message + bar while below the
 * threshold, an unlocked message once met. Renders nothing when no threshold applies.
 *
 * @param props.state - Resolved free-shipping state from `resolveFreeShipping`.
 * @param props.currencyCode - Cart presentment currency for the remaining-amount price.
 * @param props.i18n - Locale dictionary for the cart copy.
 * @returns The messaging row, or `null` when `state.state === 'none'`.
 */
export function FreeShippingProgress({ state, currencyCode, i18n }: FreeShippingProgressProps) {
    const { t } = getTranslations('cart', i18n);

    if (state.state === 'none') {
        return null;
    }

    if (state.state === 'unlocked') {
        return (
            <div className="font-semibold text-(--text-success-strong) text-sm">
                {t('free-shipping-on-this-order')}
            </div>
        );
    }

    const progress = Math.min(1, (state.threshold - state.remaining) / state.threshold);

    return (
        <div className="flex flex-col gap-1">
            <div className="text-sm leading-snug">
                {t('away-from-free-shipping', <Price key="remaining" data={{ currencyCode, amount: state.remaining.toString() }} />)}
            </div>
            <div className="h-1.5 w-full overflow-hidden rounded-full bg-(--color-gray)">
                <div
                    className={cn('h-full rounded-full bg-(--surface-success) transition-all duration-300')}
                    style={{ width: `${Math.round(progress * 100)}%` }}
                />
            </div>
        </div>
    );
}
```
