getArticle

● cms · /api · async function throws NotFoundError · returns nullable
Fetch one article by slug for a tenant. Tenant-scoped on shop.id, depth 2, limit 1. Returns Article when published, or null when the slug doesn't resolve. Throws NotFoundError on a structurally invalid slug. See Packages › cms › API for the wider read-API narrative.

Signature#

packages/cms/src/api/get-article.ts 
export async function getArticle(opts: GetArticleArgs): Promise<Article | null>;

export type GetArticleArgs = {
  shop: ShopRef;
  locale: LocaleRef;
  slug: string;
  draft?: boolean;
};

Parameters#

NameTypeDescription
opts.shop
ShopRef
Tenant reference — must include id. Asserted at call time via assertShopId.
opts.locale
LocaleRef
Request locale. Falls back to the shop's i18n.defaultLocale when the article isn't translated in the request locale.
opts.slug
string
URL slug of the article (the path segment, not the title).
opts.draft ?
boolean
When true, returns unpublished drafts. Editor-only.

Returns#

Article | null
union
The matching article, or null when no published article exists for that slug in the requested locale.

Throws#

NotFoundError when the slug is structurally invalid — empty, contains slashes, or fails InvalidHandleError's pattern check upstream.

Example#

apps/storefront/app/[domain]/[locale]/news/[slug]/page.tsx 
import { getArticle } from '@nordcom/commerce-cms/api';
import { notFound } from 'next/navigation';

export default async function ArticlePage({ params }: { params: Promise<{ shop: ShopDomain; locale: LocaleCode; slug: string }> }) {
  const { shop, locale, slug } = await params;
  const article = await getArticle({ shop, locale, slug });

  if (!article) return notFound();
  return <ArticleView article={article} />;
}
