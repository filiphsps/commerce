export async function middleware(req: NextRequest) {
  const hostname = req.headers.get('host') ?? '';
  const shop = await resolveShopFromHostname(hostname);
  if (!shop) throw new UnknownShopDomainError(hostname);

  const locale = resolveLocale(req, shop);
  return NextResponse.rewrite(new URL(`/${shop.domain}/${locale}${req.nextUrl.pathname}`, req.url));
}

Diff

packages/cms/src/api/get-article.ts 
export async function getArticle(opts: GetArticleArgs) {
  const payload = await getPayloadInstance();
  const payload = opts.__payload ?? await getPayloadInstance();
  return payload.find({
    where: { slug: { equals: opts.slug } },
    where: { and: [{ tenant: { equals: opts.shop.id } }, { slug: { equals: opts.slug } }] },
    locale: opts.locale.code,
  });
}
Diff fences via [!code ++] / [!code --] Shiki transformers. Green = add, red = del. Add lines full opacity; del lines 0.7 opacity to weight reader's eye toward the new state.
